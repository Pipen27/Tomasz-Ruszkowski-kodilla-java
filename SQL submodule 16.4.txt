SELECT i.SUMMARY, l.NAME
FROM issue i 
JOIN issueslists l  ON l.ID = i.ISSUESLIST_ID;

SELECT i.SUMMARY, u.LASTNAME, u.FIRSTNAME
FROM issue i 
JOIN users u  ON u.ID = i.USER_ID_ASSIGNEDTO
order by u.LASTNAME;

SELECT u.LASTNAME, u.FIRSTNAME, i.USER_ID_ASSIGNEDTO, count(*) as ISSUES_QUANTITY
FROM issue i 
JOIN users u  ON u.ID = i.USER_ID_ASSIGNEDTO
group by i.USER_ID_ASSIGNEDTO
order by u.LASTNAME;
